module L4S1(clr,clk,ina,inb,inc,ind,f,times,cout);

input clr,clk,ina,inb,inc,ind;
output reg f,cout;
output reg [3:0] times;

always @(posedge clk)
begin
        if (times<=10) 
        begin 
            cout = 1;
            if(times==10)
            begin
                cout = 0;
            end
            else times=times + 1;
        end
end

module Cnt59(clr,clk,q,cout);        //异步进位
input clr,clk;
output reg [6:0] q;
output reg cout;
always @(posedge clk or posedge clr)
begin
    if(clr) q=0;
        else begin
    if(q==58) q=0;
        else q=q+1;
    if(q==0) cout=1;
        else cout=0;end
end
endmodule
