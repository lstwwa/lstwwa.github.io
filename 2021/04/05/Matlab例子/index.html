legend('y1'); 

 title('sinc');

plot(t2,y2);

单位冲击响应，阶跃响应

num=1;

den=[1 3 2];

sys=tf(num,den);

t=0:0.01:6;

ht=impulse(sys,t);

gt=step(sys,t);

ha=exp(-t)-exp(-2*t);

ga=0.05-exp(-t)+0.5*exp(-2*t);

subplot(211);

plot(t,ht,' -',t,ha,' -.');

box off;

xlabel('(a)-冲击响应');

subplot(212);

plot(t,gt,' -',t,ga,' -.');

xlabel('(b)-阶跃响应');

box off;

零状态响应
